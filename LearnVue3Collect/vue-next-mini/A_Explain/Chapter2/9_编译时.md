### 利用 compile createApp 渲染页面

```html
<head>
  <script src="vueCDNUrl"></script>
</head>
<body>
  <div id="app"></div>
</body>

<script>
  const { compile createApp } = Vue

  const html = `
    <div class="test">hello compiler</div>
  `

  // renderFn 是一个函数
  const renderFn = compile(html)

  const app = createApp({
    render: renderFn
  })
  
  app.mount('#app')
  
</script>
```
